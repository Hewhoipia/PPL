
        func bubbleSort(number arr[10], number n)
        begin
            number i <- 0
            for i until i >= n - 1 by 1
            begin
                number j <- 0
                for j until j >= n - 1 - i by 1
                begin
                    if (arr[j] > arr[j+1]) begin
                        number temp <- arr[j]
                        arr[j] <- arr[j+1]
                        arr[j+1] <- temp
                    end
                end
            end
        end

        func print1DArray(number arr[10], number n)
        begin
            number i <- 0
            for i until i > n - 1 by 1
            begin
                writeNumber(arr[i])
                if (i = n - 1) break
                writeString(" ")
            end
        end
