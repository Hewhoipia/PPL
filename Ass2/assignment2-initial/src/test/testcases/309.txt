begin
    return a + b
end
func sub(number a, number b) 
begin
    return a - b
end
func mul(number a, number b) 
begin
    return a * b
end
func div(number a, number b) 
begin
    if (b != 0) 
    begin
        return a / b
    end
    else 
    begin
        return 0
    end
end
